定位：跨 JVM 的全局编程锁， 解决多进程 / 多 JVM / 跨服务并发冲突的全局锁
实现方案
（1）基于数据库的分布式锁（最简单，但性能最差）
    -- 锁表结构
    CREATE TABLE `distributed_lock` (
      `id` bigint NOT NULL AUTO_INCREMENT,
      `lock_key` varchar(64) NOT NULL COMMENT '锁标识',
      `holder` varchar(64) NOT NULL COMMENT '锁持有者（比如服务IP+线程ID）',
      `expire_time` datetime NOT NULL COMMENT '锁过期时间',
      PRIMARY KEY (`id`),
      UNIQUE KEY `uk_lock_key` (`lock_key`) COMMENT '唯一索引保证锁原子性'
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    // 获取锁
    public boolean acquireLock(String lockKey, String holder, long expireSeconds) {
        String sql = "INSERT INTO distributed_lock (lock_key, holder, expire_time) VALUES (?, ?, DATE_ADD(NOW(), INTERVAL ? SECOND))";
        try {
            jdbcTemplate.update(sql, lockKey, holder, expireSeconds);
            return true; // 插入成功=获取锁
        } catch (DuplicateKeyException e) {
            // 锁已被占用，清理过期锁后重试
            cleanExpiredLock(lockKey);
            return false;
        }
    }

    // 释放锁
    public boolean releaseLock(String lockKey, String holder) {
        String sql = "DELETE FROM distributed_lock WHERE lock_key = ? AND holder = ?";
        int rows = jdbcTemplate.update(sql, lockKey, holder);
        return rows > 0;
    }
(2) redis 分布式锁
    实现：Redis SET NX PX

    优点：性能极高（内存操作，QPS 可达 10 万 +）
    缺点：依赖redis集群，非持久化（Redis 重启会丢失锁，需开启 AOF）， 且极端场景下 RedLock 仍有一致性风险
    锁过期：Redisson 的「看门狗机制」—— 每隔 10 秒自动给锁续期（默认续到 30 秒），直到业务执行完成。
    误删锁问题：线程 A 的锁过期后，线程 B 获取锁，线程 A 执行完后误删线程 B 的锁；
    ✅ 解决方案：解锁时用 Lua 脚本原子判断 holder（锁持有者）：
        if redis.call('get', KEYS[1]) == ARGV[1] then
            return redis.call('del', KEYS[1])
        else
            return 0
        end
    死锁问题：线程获取锁后宕机，锁无法释放；
    ✅ 解决方案：设置锁过期时间，即使宕机，锁也会自动释放。
    核心避坑点
        1. 必须保证「锁的原子性」
            加锁 / 解锁操作必须是原子的（比如 Redis 的 SET NX PX、数据库的唯一索引）；
            反例：先判断 Key 是否存在，再设置 Key（非原子，高并发下会导致多个线程同时获取锁）。
        2. 必须保证「锁的归属权」
            解锁时必须验证锁持有者（比如 Redis 的 holder、数据库的 holder 字段），避免误删其他线程的锁。
        3. 必须设置「锁过期时间」
            无论哪种实现，都要设置过期时间，避免线程宕机导致死锁。
        4. 必须保证「幂等性」
            分布式锁只能保证并发执行的串行化，无法解决「网络重试导致的重复执行」；
            比如：扣库存接口被重试，即使加了分布式锁，也可能重复扣减，需额外加幂等性校验（如订单号 + 操作类型）。
    核心特性（Redisson 实现）
        特性	说明
        可重入	同一线程可多次获取同一把锁（基于 ThreadLocal）
        公平锁	按线程等待顺序获取锁（基于 Redis 有序集合）
        自动续期	看门狗机制，避免锁提前过期
        超时获取	支持 tryLock (timeout)，超时放弃获取锁
        异步锁	支持异步获取锁（asyncLock），提升性能

（3）基于 ZooKeeper/ETCD 的分布式锁（最可靠，性能中等）
    实现：利用 ZooKeeper 的临时有序节点：
       线程在/lock/目录下创建临时有序节点（如/lock/lock_key-000000001）；
       判断自己是否是最小节点，是则获取锁，否则监听前一个节点；
       锁释放时，前一个节点删除，当前节点被唤醒，重新判断。
    优点：天然支持分布式一致性（CP 模型）、锁自动释放（临时节点）、无过期时间
    缺点：性能低于 Redis（ZooKeeper 是 CP，强一致但性能低）、部署维护复杂（需 ZooKeeper 集群）、高并发下ZK压力大


Redission实现示例
    import org.redisson.api.RLock;
    import org.redisson.api.RedissonClient;
    import org.springframework.beans.factory.annotation.Autowired;
    import org.springframework.stereotype.Component;

    import java.util.concurrent.TimeUnit;

    /**
     * Redisson实现Redis分布式锁（推荐）
     */
    @Component
    public class RedissonDistributedLock {

        @Autowired
        private RedissonClient redissonClient;

        /**
         * 获取分布式锁（带超时等待+自动续期）
         * @param lockKey 锁标识（如：stock_lock_1001）
         * @param waitTime 等待获取锁的超时时间（秒）
         * @param leaseTime 锁的初始过期时间（秒，Redisson看门狗会自动续期）
         * @return 锁对象（用于释放锁）
         * @throws InterruptedException 等待被中断时抛出
         */
        public RLock acquireLock(String lockKey, long waitTime, long leaseTime) throws InterruptedException {
            // 获取锁对象
            RLock lock = redissonClient.getLock(lockKey);
            // 加锁：waitTime秒内获取不到则失败，获取成功后初始租期leaseTime秒（看门狗自动续期）
            boolean locked = lock.tryLock(waitTime, leaseTime, TimeUnit.SECONDS);
            if (!locked) {
                throw new RuntimeException("获取分布式锁失败，锁Key：" + lockKey);
            }
            return lock;
        }

        /**
         * 释放分布式锁
         * @param lock 锁对象
         */
        public void releaseLock(RLock lock) {
            if (lock != null && lock.isHeldByCurrentThread()) {
                // 仅释放当前线程持有的锁，避免误删
                lock.unlock();
            }
        }

        // 业务使用示例
        public void deductStock(Long productId) {
            String lockKey = "stock_lock_" + productId;
            RLock lock = null;
            try {
                // 获取锁：最多等5秒，初始租期30秒（看门狗每10秒续期一次）
                lock = acquireLock(lockKey, 5, 30);
                // ------------------------
                // 核心业务逻辑：扣减库存
                System.out.println("线程" + Thread.currentThread().getId() + "获取锁，开始扣减商品" + productId + "库存");
                Thread.sleep(5000); // 模拟业务执行时间
                System.out.println("线程" + Thread.currentThread().getId() + "扣减库存完成");
                // ------------------------
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                throw new RuntimeException("扣减库存中断", e);
            } finally {
                // 释放锁（必须在finally中，避免死锁）
                releaseLock(lock);
            }
        }
    }
    关键特性说明
        自动续期（看门狗）：如果业务执行时间超过 30 秒，Redisson 会自动每隔 10 秒将锁续期到 30 秒，直到业务执行完成；
        可重入：同一线程可多次调用tryLock获取同一把锁，不会死锁；
        安全解锁：isHeldByCurrentThread()确保只释放当前线程持有的锁，避免误删其他线程的锁；
        超时控制：tryLock(waitTime, leaseTime)支持 “等待超时” 和 “锁过期” 双重控制。

lua脚本实现：setIfAbsent 获取锁 + lua脚本释放锁（原子）
    import org.springframework.beans.factory.annotation.Autowired;
    import org.springframework.data.redis.core.RedisTemplate;
    import org.springframework.data.redis.core.script.DefaultRedisScript;
    import org.springframework.stereotype.Component;

    import java.util.Collections;
    import java.util.UUID;
    import java.util.concurrent.TimeUnit;

    /**
     * 手动Lua脚本实现Redis分布式锁（底层实现）
     */
    @Component
    public class LuaRedisDistributedLock {

        @Autowired
        private RedisTemplate<String, String> redisTemplate;

        // 解锁的Lua脚本（原子判断+删除，避免误删）
        private static final String UNLOCK_LUA_SCRIPT = """
                if redis.call('get', KEYS[1]) == ARGV[1] then
                    return redis.call('del', KEYS[1])
                else
                    return 0
                end
                """;
        private final DefaultRedisScript<Long> unlockScript;

        public LuaRedisDistributedLock() {
            // 初始化解锁脚本
            unlockScript = new DefaultRedisScript<>();
            unlockScript.setScriptText(UNLOCK_LUA_SCRIPT);
            unlockScript.setResultType(Long.class);
        }

        /**
         * 获取分布式锁（手动实现，无自动续期）
         * @param lockKey 锁标识
         * @param expireSeconds 锁过期时间（秒，必须大于业务执行时间）
         * @return 锁持有者ID（用于解锁），null表示获取失败
         */
        public String acquireLock(String lockKey, long expireSeconds) {
            // 生成唯一的锁持有者ID（避免误删其他线程的锁）
            String holderId = UUID.randomUUID().toString();
            // 加锁指令：SET lockKey holderId NX PX expireSeconds*1000
            // NX=仅当Key不存在时设置，PX=毫秒级过期时间
            Boolean success = redisTemplate.opsForValue()
                    .setIfAbsent(lockKey, holderId, expireSeconds, TimeUnit.SECONDS);
            // 成功返回holderId，失败返回null
            return Boolean.TRUE.equals(success) ? holderId : null;
        }

        /**
         * 释放分布式锁（Lua脚本原子操作）
         * @param lockKey 锁标识
         * @param holderId 锁持有者ID（acquireLock返回的值）
         * @return 是否释放成功
         */
        public boolean releaseLock(String lockKey, String holderId) {
            if (holderId == null) {
                return false;
            }
            // 执行Lua脚本，原子判断并删除锁
            Long result = redisTemplate.execute(
                    unlockScript,
                    Collections.singletonList(lockKey), // KEYS[1] = lockKey
                    holderId // ARGV[1] = holderId
            );
            // result=1表示解锁成功，0表示锁已过期/不属于当前线程
            return result != null && result == 1;
        }

        // 业务使用示例
        public void deductStock(Long productId) {
            String lockKey = "stock_lock_" + productId;
            String holderId = null;
            try {
                // 获取锁：过期时间设为60秒（必须大于业务执行时间）
                holderId = acquireLock(lockKey, 60);
                if (holderId == null) {
                    throw new RuntimeException("获取分布式锁失败，锁Key：" + lockKey);
                }
                // ------------------------
                // 核心业务逻辑：扣减库存
                System.out.println("线程" + Thread.currentThread().getId() + "获取锁，开始扣减商品" + productId + "库存");
                Thread.sleep(5000); // 模拟业务执行时间
                System.out.println("线程" + Thread.currentThread().getId() + "扣减库存完成");
                // ------------------------
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                throw new RuntimeException("扣减库存中断", e);
            } finally {
                // 释放锁
                if (holderId != null) {
                    boolean released = releaseLock(lockKey, holderId);
                    System.out.println("锁释放结果：" + released);
                }
            }
        }
    }

Redisson 是 Redis 分布式锁的 Java 实现框架，而 RedLock 是一种解决 Redis 集群下分布式锁一致性问题的算法
    基于 Redis 的 Java 客户端框架（分布式锁只是其功能之一）

Redisson RedLock（红锁算法）
    核心规则（Redis 官方定义）：
    向所有独立 Redis 节点（至少 5 个）发送SET lock_key holder NX PX 30000指令；
    计算获取锁的成功节点数，超过半数（≥3）且总耗时≤锁过期时间的一半，则认为锁获取成功；
    若获取失败，向所有节点发送解锁指令（即使未获取到锁的节点）；
    锁释放时，向所有节点发送解锁指令；
    优势：只要超过半数节点存活，锁就不会失效，解决主从切换的一致性问题；

RedLock 并非 “银弹”，存在以下争议和局限：
    时钟漂移问题：若某个 Redis 节点的系统时钟发生跳跃，可能导致锁提前过期，破坏一致性；
    性能损耗：需访问所有节点，网络开销是普通锁的 5 倍以上，高并发下性能下降明显；
    部署成本高：需维护至少 5 个独立 Redis 节点（不能是主从 / 集群，必须独立），运维成本高；
    Redis 官方态度：Redis 作者认为 RedLock 仅适合 “非金融级” 强一致场景，金融级场景建议用 ZooKeeper。
